            int main () {
                char MAIN[100], PAT[20], REP[20];
                printf("Enter the main string:");
                gets(MAIN);
                printf("Enter the PATtern string:");
                gets(PAT);
                printf("Enter the replace string:");
                gets(REP);
                int flag = 0, j;
                int len = strlen(MAIN);
                int pLen = strlen(PAT);
                for (int i = 0; i <= len - pLen; i++) {
                    if (MAIN[i] == PAT[0]) {
                        for (j = 0; j < pLen; j++) {
                            if (MAIN[i + j] != PAT[j]) break;
                        }
                        if (j == pLen) {
                            for (int j = 0; j < pLen; j++) {
                                MAIN[i + j] = REP[j];
                            }
                            i += pLen;
                            flag = 1;
                        }
                    }
                }
                if (flag == 1) printf("%s", MAIN);
                else printf("Pattern not found");
                return 0;
            }
