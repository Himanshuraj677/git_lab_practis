            #define SIZE 10
            int stack[SIZE]; //Declare a fixed size array to use as the stack
            int top = -1;    //Initialize
            void push (int data) {
                if (top == SIZE - 1) {
                    printf("Stack Overflow\n");
                    return;
                }
                stack[++top] = data;
            }
            
            void pop () {
                if (top == -1) {
                    printf("Stack Underflow\n");
                    return ;
                }
                printf("%d get pop\n",stack[top--]);
            }
            
            void isPalindrome () {
                int i = 0, j = top;
                while (i <= top / 2 ) {
                    if (stack[i] != stack[j]){
                        printf("Not Palindrome \n");
                        return;
                    }
                    i++;
                    j--;
                }
                printf("It's a palindrome!\n");
            }
            
            void display () {
                int i = 0;
                for(i=0; i<=top; i++)
                    printf("%d ",stack[i]);
                printf("\n");
            }
            
            int main () {
                printf("1) Push\n2) Pop\n3) isPalindrome\n4) Display\n5) Exit\n");
                int choice;
                while (1) {
                    printf("Enter your choice:");
                    scanf("%d", &choice);
                    switch (choice)
                    {
                    case 1:
                        printf("Enter the num to be pushed:");
                        int data;
                        scanf("%d",&data);
                        push(data);
                        break;
                    case 2:
                        pop();
                        break;
                    case 3:
                        isPalindrome();
                        break;
                    case 4:
                        display();
                        break;
                    case 5:
                        exit(0);
            
                    default:
                        printf("You have entered wrong choice\n");
                    }
                }
                return 0;
            }
