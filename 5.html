            #define MAX 100
            int stack[100];
            int top = -1;
            
            void push(char x) {
                stack[++top] = x;
            }
            
            char pop() {
                return stack[top--];
            }
            
            int calculate (int num1, int num2, char op) {
                switch (op)
                {
                case '+': return num1 + num2;
                case '-': return num1 - num2;
                case '*': return num1 * num2;
                case '/': return num1 / num2;
                case '%': return num1 % num2;
                case '^': return pow(num1, num2);
                }
            }
            int main () {
                char psfx[100];
                printf("Enter your postfix expression:");
                scanf("%[^/n]", &psfx);
                int ans;
                for (int i = 0; i < strlen(psfx); i++) {
                    char num = psfx[i];
                    if (num == ' ') continue;
                    if (isalnum(num)) push(num - '0');
                    else {
                        int op1 = pop();
                        int op2 = pop();
                        int ans = calculate(op2, op1, num);
                        push(ans);
                    }
                }
                printf("Answer is: %d", pop());
                return 0;
            }
